15  package com.codebox.builders;
16  
17  import java.util.HashMap;
18  import java.util.Map;
19  import java.util.Map.Entry;
20  
21  import javassist.CannotCompileException;
22  import javassist.ClassPool;
23  import javassist.CtClass;
24  import javassist.CtField;
25  import javassist.CtMethod;
26  import javassist.NotFoundException;
27  
28  /**
29   * The Class ExtensionBuilder.
30   *
31   * @param <T>
32   *            the generic type
33   */
34  public class ExtensionBuilder<T> {
35  
36      /**
37       * Generate.
38       *
39       * @param clazz
40       *            the clazz
41       * @return the class
42       * @throws NotFoundException
43       *             the not found exception
44       * @throws CannotCompileException
45       *             the cannot compile exception
46       */
47      public Class<?> generate(Class<T> clazz) throws NotFoundException, CannotCompileException {
48          try {
49              // If extension already recreated, return it
50              return Class.forName(clazz.getName() + "Extension");
51          } catch (@SuppressWarnings("unused") ClassNotFoundException e) {
52              // No extension exists, so create it
53          }
54  
55          final ClassPool pool = ClassPool.getDefault();
56          final CtClass cc = pool.makeClass(clazz.getName() + "Extension");
57  
58          // add super class
59          cc.setSuperclass(resolveCtClass(clazz));
60  
61          final Map<String, Class<?>> properties = new HashMap<String, Class<?>>();
62          properties.put("jbExtension1", String.class);
63          properties.put("jbExtension2", String.class);
64          properties.put("jbExtension3", String.class);
65          properties.put("jbExtension4", String.class);
66  
67          for (final Entry<String, Class<?>> entry : properties.entrySet()) {
68  
69              // Add field
70              cc.addField(new CtField(resolveCtClass(entry.getValue()), entry.getKey(), cc));
71  
72              // Add getter
73              cc.addMethod(generateGetter(cc, entry.getKey(), entry.getValue()));
74  
75              // Add setter
76              cc.addMethod(generateSetter(cc, entry.getKey(), entry.getValue()));
77          }
78  
79          return cc.toClass();
80      }
81  
82      /**
83       * Generate getter.
84       *
85       * @param declaringClass
86       *            the declaring class
87       * @param fieldName
88       *            the field name
89       * @param fieldClass
90       *            the field class
91       * @return the ct method
92       * @throws CannotCompileException
93       *             the cannot compile exception
94       */
95      private static CtMethod generateGetter(CtClass declaringClass, String fieldName, Class<?> fieldClass)
96              throws CannotCompileException {
97          final StringBuilder sb = new StringBuilder();
98          sb.append("public ").append(fieldClass.getName()).append(" get")
99                  .append(fieldName.substring(0, 1).toUpperCase()).append(fieldName.substring(1)).append("(){")
100                 .append("return this.").append(fieldName).append(";").append("}");
101         return CtMethod.make(sb.toString(), declaringClass);
102     }
103 
104     /**
105      * Generate setter.
106      *
107      * @param declaringClass
108      *            the declaring class
109      * @param fieldName
110      *            the field name
111      * @param fieldClass
112      *            the field class
113      * @return the ct method
114      * @throws CannotCompileException
115      *             the cannot compile exception
116      */
117     private static CtMethod generateSetter(CtClass declaringClass, String fieldName, Class<?> fieldClass)
118             throws CannotCompileException {
119         final StringBuilder sb = new StringBuilder();
120         sb.append("public void set").append(fieldName.substring(0, 1).toUpperCase()).append(fieldName.substring(1))
121                 .append("(").append(fieldClass.getName()).append(" ").append(fieldName).append(")").append("{")
122                 .append("this.").append(fieldName).append("=").append(fieldName).append(";").append("}");
123         return CtMethod.make(sb.toString(), declaringClass);
124     }
125 
126     /**
127      * Resolve ct class.
128      *
129      * @param clazz
130      *            the clazz
131      * @return the ct class
132      * @throws NotFoundException
133      *             the not found exception
134      */
135     private static CtClass resolveCtClass(Class<?> clazz) throws NotFoundException {
136         final ClassPool pool = ClassPool.getDefault();
137         return pool.get(clazz.getName());
138     }
139 
140 }
